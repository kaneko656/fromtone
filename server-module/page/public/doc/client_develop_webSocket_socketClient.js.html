// socket

/**
 * @param  {callback}
 */

exports.connecting = (callback = () => {}) => {
    if (isConnecting) {
        callback(url)
    }
}

/**
 * @param  {callback}
 */

exports.connect = (callback = () => {}) => {
    if (isConnect) {
        callback(url)
    }
    call.on('connect', () => {
        isConnecting = false
        callback(url)
    })
}
let connect = exports.connect


/**
 * @param  {callback}
 */

exports.disconnect = (callback = () => {}) => {
    if (!isConnect && !isConnecting) {
        callback(url)
    }
    call.on('disconnect', () => {
        callback(url)
    })
}
let disconnect = exports.disconnect
